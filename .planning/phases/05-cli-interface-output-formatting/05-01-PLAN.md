   ```javascript
   function formatCompactTime(ms) {
     if (ms <= 0) return '0s';
     const days = Math.floor(ms / 86400000);
     const hours = Math.floor((ms % 86400000) / 3600000);
     const minutes = Math.floor((ms % 3600000) / 60000);
     const seconds = Math.floor((ms % 60000) / 1000);
     const parts = [];
     if (days > 0) parts.push(`${days}d`);
     if (hours > 0) parts.push(`${hours}h`);
     if (minutes > 0) parts.push(`${minutes}m`);
     if (seconds > 0 && days === 0) parts.push(`${seconds}s`);
     return parts.join('');
   }
   ```

2. **Fix normalizeResetTime** (line 882): Add millisecond detection heuristic.
   Change:
   ```javascript
   resetDate = new Date(resetTime * 1000);
   ```

   ```javascript
   // Heuristic: >1e12 is already milliseconds, <1e12 is seconds
   resetDate = new Date(resetTime > 1e12 ? resetTime : resetTime * 1000);
   ```
